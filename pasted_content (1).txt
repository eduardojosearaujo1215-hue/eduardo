    <script>
        // Inicializar ícones
        lucide.createIcons();

        // --- Configuração Global ---
        let db;
        let scannerIniciado = false;
        const html5QrCode = new Html5Qrcode("reader");

        // --- Inicialização do Banco de Dados ---
        const request = indexedDB.open("EstoqueDB", 2); 
        request.onupgradeneeded = (e) => {
            db = e.target.result;
            if (db.objectStoreNames.contains("produtos")) {
                db.deleteObjectStore("produtos");
            }
            const store = db.createObjectStore("produtos", { keyPath: "id", autoIncrement: true });
            store.createIndex("barcode", "barcode", { unique: false });
        };
        request.onsuccess = (e) => {
            db = e.target.result;
            console.log("Banco de dados pronto");
            iniciarScanner();
        };

        // --- Navegação de Abas ---
        function switchTab(tab) {
            document.querySelectorAll('.tab-content').forEach(el => el.classList.remove('active'));
            document.querySelectorAll('nav button').forEach(el => {
                el.classList.remove('active-tab', 'text-blue-600');
                el.classList.add('text-gray-400');
            });

            document.getElementById('tab-' + tab).classList.add('active');
            const navBtn = document.getElementById('nav-' + tab);
            navBtn.classList.add('active-tab', 'text-blue-600');
            navBtn.classList.remove('text-gray-400');

            if (tab === 'estoque') listarProdutos();
            if (tab === 'precificacao') carregarProdutosSelect();
        }

        // --- Funções de Câmera ---
        async function startCamera() {
            const video = document.getElementById('webcam');
            const btnCamera = document.getElementById('btn-camera');
            const btnTakePhoto = document.getElementById('btn-take-photo');
            const emptyPhoto = document.getElementById('empty-photo');
            const preview = document.getElementById('photo-preview');

            try {
                const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: 'environment' } });
                video.srcObject = stream;
                video.classList.remove('hidden');
                btnTakePhoto.classList.remove('hidden');
                btnCamera.classList.add('hidden');
                emptyPhoto.classList.add('hidden');
                preview.classList.add('hidden');
            } catch (err) {
                alert("Erro ao acessar câmera: " + err);
            }
        }

        function takePhoto() {
            const video = document.getElementById('webcam');
            const canvas = document.getElementById('canvas');
            const preview = document.getElementById('photo-preview');
            const btnCamera = document.getElementById('btn-camera');
            const btnTakePhoto = document.getElementById('btn-take-photo');

            canvas.width = video.videoWidth;
            canvas.height = video.videoHeight;
            canvas.getContext('2d').drawImage(video, 0, 0);
            
            const dataUrl = canvas.toDataURL('image/jpeg', 0.7);
            preview.src = dataUrl;
            preview.classList.remove('hidden');
            video.classList.add('hidden');
            btnTakePhoto.classList.add('hidden');
            btnCamera.classList.remove('hidden');
            btnCamera.innerHTML = '<i data-lucide="refresh-cw" class="w-5 h-5"></i> Tirar Outra';
            
            video.srcObject.getTracks().forEach(track => track.stop());
            lucide.createIcons();
        }

        // --- Funções de Scanner ---
        function iniciarScanner() {
            if (scannerIniciado) return;
            const config = { fps: 10, qrbox: { width: 250, height: 150 } };
            html5QrCode.start({ facingMode: "environment" }, config, onScanSuccess);
            scannerIniciado = true;
        }

        function onScanSuccess(decodedText) {
            document.getElementById('prod-barcode').value = decodedText;
            buscarProdutoPorBarcode(decodedText);
        }

        function buscarProdutoPorBarcode(barcode) {
            const transaction = db.transaction(["produtos"], "readonly");
            const store = transaction.objectStore("produtos");
            const index = store.index("barcode");
            index.get(barcode).onsuccess = (e) => {
                const p = e.target.result;
                if (p) {
                    document.getElementById('prod-nome').value = p.nome;
                    if(p.foto) {
                        const preview = document.getElementById('photo-preview');
                        preview.src = p.foto;
                        preview.classList.remove('hidden');
                        document.getElementById('empty-photo').classList.add('hidden');
                    }
                }
            };
        }

        // --- CRUD ---
        function salvarProduto() {
            const id = document.getElementById('prod-id').value;
            const barcode = document.getElementById('prod-barcode').value;
            const nome = document.getElementById('prod-nome').value;
            const qtd = parseInt(document.getElementById('prod-qtd').value) || 0;
            const validade = document.getElementById('prod-validade').value;
            const foto = document.getElementById('photo-preview').classList.contains('hidden') ? null : document.getElementById('photo-preview').src;

            if (!nome || !barcode) return alert("Preencha Nome e Código de Barras!");

            const produto = { 
                barcode, 
                nome, 
                qtd, 
                validade, 
                foto,
                precoCompra: 0,
                precoVenda: 0,
                precoSugestao: 0
            };

            const transaction = db.transaction(["produtos"], "readwrite");
            const store = transaction.objectStore("produtos");

            if (id) {
                store.get(parseInt(id)).onsuccess = (e) => {
                    const original = e.target.result;
                    produto.id = parseInt(id);
                    produto.precoCompra = original.precoCompra || 0;
                    produto.precoVenda = original.precoVenda || 0;
                    produto.precoSugestao = original.precoSugestao || 0;
                    store.put(produto);
                    finalizarSalvar();
                };
            } else {
                store.add(produto);
                finalizarSalvar();
            }
        }

        function finalizarSalvar() {
            alert("Produto salvo com sucesso!");
            limparForm();
            if (document.getElementById('tab-estoque').classList.contains('active')) listarProdutos();
        }

        function editarProduto(id) {
            const transaction = db.transaction(["produtos"], "readonly");
            const store = transaction.objectStore("produtos");
            store.get(id).onsuccess = (e) => {
                const p = e.target.result;
                document.getElementById('prod-id').value = p.id;
                document.getElementById('prod-barcode').value = p.barcode;
                document.getElementById('prod-nome').value = p.nome;
                document.getElementById('prod-qtd').value = p.qtd;
                document.getElementById('prod-validade').value = p.validade;
                
                if (p.foto) {
                    const preview = document.getElementById('photo-preview');
                    preview.src = p.foto;
                    preview.classList.remove('hidden');
                    document.getElementById('empty-photo').classList.add('hidden');
                }

                document.getElementById('btn-salvar').textContent = "ATUALIZAR PRODUTO";
                document.getElementById('btn-cancelar').classList.remove('hidden');
                switchTab('cadastro');
            };
        }

        function excluirProduto(id) {
            if (confirm("Deseja excluir este item do estoque?")) {
                const transaction = db.transaction(["produtos"], "readwrite");
                const store = transaction.objectStore("produtos");
                store.delete(id).onsuccess = () => listarProdutos();
            }
        }

        function limparForm() {
            document.getElementById('prod-id').value = '';
            document.getElementById('prod-barcode').value = '';
            document.getElementById('prod-nome').value = '';
            document.getElementById('prod-qtd').value = '';
            document.getElementById('prod-validade').value = '';
            document.getElementById('photo-preview').src = '';
            document.getElementById('photo-preview').classList.add('hidden');
            document.getElementById('empty-photo').classList.remove('hidden');
            document.getElementById('btn-salvar').textContent = "SALVAR NO ESTOQUE";
            document.getElementById('btn-cancelar').classList.add('hidden');
            document.getElementById('btn-camera').innerHTML = '<i data-lucide="camera" class="w-5 h-5"></i> Abrir Câmera';
            lucide.createIcons();
        }

        // --- Precificação ---
        function carregarProdutosSelect() {
            const select = document.getElementById('price-select-prod');
            select.innerHTML = '<option value="">Selecione um produto...</option>';
            
            const transaction = db.transaction(["produtos"], "readonly");
            const store = transaction.objectStore("produtos");
            const produtos = [];

            store.openCursor().onsuccess = (e) => {
                const cursor = e.target.result;
                if (cursor) {
                    produtos.push(cursor.value);
                    cursor.continue();
                } else {
                    // Ordenar do mais novo para o mais antigo (ID decrescente)
                    produtos.sort((a, b) => (b.id || 0) - (a.id || 0));
                    
                    produtos.forEach(p => {
                        const option = document.createElement('option');
                        option.value = p.id;
                        option.textContent = `${p.nome} (${p.barcode})`;
                        select.appendChild(option);
                    });

                    // Selecionar automaticamente o último cadastrado (primeiro da lista ordenada)
                    if (produtos.length > 0) {
                        select.value = produtos[0].id;
                        carregarDadosPreco();
                    }
                }
            };
        }

        function calcularUnitario() {
            const qtd = parseFloat(document.getElementById('calc-qtd').value) || 0;
            const total = parseFloat(document.getElementById('calc-total').value) || 0;
            if (qtd > 0 && total > 0) {
                document.getElementById('price-compra').value = (total / qtd).toFixed(2);
                calcularMargem();
            }
        }

        function calcularMargem() {
            const compra = parseFloat(document.getElementById('price-compra').value) || 0;
            const venda = parseFloat(document.getElementById('price-venda').value) || 0;
            const margemLabel = document.getElementById('margem-valor');
            const sugestaoArea = document.getElementById('sugestao-area');

            if (venda > 0) {
                const margem = ((venda - compra) / venda) * 100;
                margemLabel.textContent = margem.toFixed(1) + "%";
                
                if (margem < 20) {
                    sugestaoArea.classList.remove('hidden');
                } else {
                    sugestaoArea.classList.add('hidden');
                }
            } else {
                margemLabel.textContent = "0%";
                sugestaoArea.classList.add('hidden');
            }
        }

        function aplicarSugestao() {
            const compra = parseFloat(document.getElementById('price-compra').value) || 0;
            const margemDesejada = parseFloat(document.getElementById('input-margem-sugerida').value) || 30;
            
            if (compra > 0) {
                const novaVenda = compra / (1 - (margemDesejada / 100));
                document.getElementById('price-venda').value = novaVenda.toFixed(2);
                calcularMargem();
            }
        }

        function carregarDadosPreco() {
            const id = parseInt(document.getElementById('price-select-prod').value);
            if (!id) return;

            const transaction = db.transaction(["produtos"], "readonly");
            const store = transaction.objectStore("produtos");
            store.get(id).onsuccess = (e) => {
                const p = e.target.result;
                document.getElementById('price-compra').value = p.precoCompra || 0;
                document.getElementById('price-venda').value = p.precoVenda || 0;
                document.getElementById('price-sugestao').value = p.precoSugestao || '';
                calcularMargem();
                calcularMargemSugestao();
            }
        }

        
function calcularMargemSugestao() {
    const compra=parseFloat(document.getElementById('price-compra').value)||0;
    const sugestao=parseFloat(document.getElementById('price-sugestao').value)||0;
    const el=document.getElementById('margem-sugestao-info');
    if(!el) return;
    if(compra>0 && sugestao>0){
      const m=((sugestao-compra)/sugestao)*100;
      el.textContent='Margem da sugestão: '+m.toFixed(1)+'%';
    } else {
      el.textContent='Margem da sugestão: 0%';
    }
}

function atualizarPreco() {
            const id = parseInt(document.getElementById('price-select-prod').value);
            const compra = parseFloat(document.getElementById('price-compra').value) || 0;
            const venda = parseFloat(document.getElementById('price-venda').value) || 0;
            const sugestao = parseFloat(document.getElementById('price-sugestao').value) || 0;

            if(!id) return alert("Selecione um produto");

            const transaction = db.transaction(["produtos"], "readwrite");
            const store = transaction.objectStore("produtos");
            store.get(id).onsuccess = (e) => {
                const p = e.target.result;
                p.precoCompra = compra;
                p.precoVenda = venda;
                p.precoSugestao = sugestao;
                p.margemSugestao = sugestao>0?(((sugestao-compra)/sugestao)*100):0;
                store.put(p);
                alert("Preços e sugestão atualizados!");
            };
        }

        // --- Listagem e WhatsApp ---
        function listarProdutos() {
            const container = document.getElementById('lista-estoque');
            const totalSpan = document.getElementById('total-itens');
            const filtro = document.getElementById('filtro-estoque')?.value || 'recentes';
            container.innerHTML = '';
            
            const transaction = db.transaction(["produtos"], "readonly");
            const store = transaction.objectStore("produtos");
            const produtos = [];

            store.openCursor().onsuccess = (e) => {
                const cursor = e.target.result;
                if (cursor) {
                    const p = cursor.value;
                    p.margem = p.precoVenda > 0 ? (((p.precoVenda - p.precoCompra) / p.precoVenda) * 100) : 0;
                    produtos.push(p);
                    cursor.continue();
                } else {
                    produtos.sort((a, b) => {
                        if (filtro === 'recentes') {
                            return (b.id || 0) - (a.id || 0);
                        } else if (filtro === 'validade') {
                            if (!a.validade) return 1;
                            if (!b.validade) return -1;
                            return a.validade.localeCompare(b.validade);
                        } else if (filtro === 'produto') {
                            return a.nome.localeCompare(b.nome);
                        } else if (filtro === 'quantidade') {
                            return (a.qtd || 0) - (b.qtd || 0);
                        } else if (filtro === 'porcentagem') {
                            return (a.margem || 0) - (b.margem || 0);
                        }
                        return 0;
                    });

                    totalSpan.textContent = `${produtos.length} Itens`;
                    if (produtos.length === 0) {
                        container.innerHTML = '<div class="text-center py-10 text-gray-400"><i data-lucide="package-search" class="w-12 h-12 mx-auto mb-2 opacity-20"></i><p class="text-sm">Nenhum produto cadastrado</p></div>';
                    } else {
                        produtos.forEach(p => {
                            const margemFormatada = (p.margem || 0).toFixed(1);
                            const div = document.createElement('div');
                            div.className = "bg-white p-4 rounded-xl border border-gray-100 shadow-sm flex flex-col gap-3";
                            div.innerHTML = `
                                <div class="flex gap-4">
                                    <div class="w-20 h-20 flex-shrink-0">
                                        ${p.foto ? `<img src="${p.foto}" class="w-full h-full object-cover rounded-lg border shadow-sm">` : `<div class="w-full h-full bg-gray-50 rounded-lg border-2 border-dashed border-gray-200 flex items-center justify-center text-gray-300"><i data-lucide="image" class="w-8 h-8"></i></div>`}
                                    </div>
                                    <div class="flex-1 min-w-0">
                                        <h3 class="font-bold text-gray-800 truncate">${p.nome}</h3>
                                        <p class="text-[10px] font-mono text-gray-400 mb-1">${p.barcode}</p>
                                        <div class="flex flex-wrap gap-2">
                                            <span class="bg-blue-50 text-blue-700 text-[10px] font-bold px-2 py-0.5 rounded uppercase">Qtd: ${p.qtd}</span>
                                            <span class="bg-orange-50 text-orange-700 text-[10px] font-bold px-2 py-0.5 rounded uppercase">Val: ${formatarData(p.validade)}</span>
                                        </div>
                                    </div>
                                </div>
                                <div class="flex justify-between items-end border-t border-gray-50 pt-3">
                                    <div>
                                        <p class="text-[10px] text-gray-400 uppercase font-bold">Preço de Venda</p>
                                        <p class="text-lg font-black text-gray-800">R$ ${p.precoVenda.toFixed(2)} <span class="text-xs font-normal text-gray-400">(${margemFormatada}%)</span></p>
                                    </div>
                                    <div class="flex gap-2">
                                        <button onclick="editarProduto(${p.id})" class="bg-blue-100 text-blue-600 p-2.5 rounded-full shadow-sm transition active:scale-90">
                                            <i data-lucide="edit-3" class="w-5 h-5"></i>
                                        </button>
                                        <button onclick="excluirProduto(${p.id})" class="bg-red-100 text-red-600 p-2.5 rounded-full shadow-sm transition active:scale-90">
                                            <i data-lucide="trash-2" class="w-5 h-5"></i>
                                        </button>
                                        <button onclick="enviarWhats(${p.id})" class="bg-green-500 text-white p-2.5 rounded-full shadow-md transition active:scale-90">
                                            <i data-lucide="message-circle" class="w-5 h-5"></i>
                                        </button>
                                    </div>
                                </div>
                            `;
                            container.appendChild(div);
                        });
                    }
                    lucide.createIcons();
                }
            };
        }

        function formatarData(data) {
            if(!data) return 'N/A';
            const [ano, mes, dia] = data.split('-');
            return `${dia}/${mes}/${ano.slice(-2)}`;
        }

        function enviarWhats(id) {
            const transaction = db.transaction(["produtos"], "readonly");
            const store = transaction.objectStore("produtos");
            store.get(id).onsuccess = (e) => {
                const p = e.target.result;
                const margem = p.precoVenda > 0 ? (((p.precoVenda - p.precoCompra) / p.precoVenda) * 100).toFixed(1) : 0;
                
                // Lógica de sugestão manual
                let sugestaoStr = "";
                if(p.precoSugestao && p.precoSugestao > 0) {
                    const ms=(p.margemSugestao??(((p.precoSugestao-p.precoCompra)/p.precoSugestao)*100)).toFixed(1);
                    sugestaoStr = `\n💡 *Sugestão:* R$ ${p.precoSugestao.toFixed(2)} (${ms}%)`;
                }

                const texto = `📦 *RELATÓRIO DE PRODUTO*\n\n` +
                              `🔹 *Nome:* ${p.nome}\n` +
                              `🔹 *Quantidade:* ${p.qtd} un\n` +
                              `🔹 *Validade:* ${formatarData(p.validade)}\n` +
                              `🔹 *Preço Compra:* R$ ${p.precoCompra.toFixed(2)}\n` +
                              `🔹 *Preço Venda:* R$ ${p.precoVenda.toFixed(2)} (${margem}%)${sugestaoStr}`;
                
                const url = `https://api.whatsapp.com/send?text=${encodeURIComponent(texto)}`;
                window.open(url, '_blank');
            };
        }
